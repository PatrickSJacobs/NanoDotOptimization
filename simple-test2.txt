#!/bin/bash
#SBATCH -J myMPI
#SBATCH -o myMPI.o%j
#SBATCH -n 32
#SBATCH -N 1
#SBATCH --mail-user=pjacobs7@eagles.nccu.edu
#SBATCH --mail-type=all
#SBATCH -p skx
#SBATCH -t 00:30:00
echo "SCRIPT $PE_HOSTFILE"
module load gcc/13.2.0
module load impi/21.11
module load meep/1.28

ibrun -np 32 meep no-metal=true theta_deg=0 sy=.4 ag-dot-angle0.ctl | tee air.out;
grep flux1: air.out > air.dat;
ibrun -np 32 meep no-metal=false sr=.04 ht=.050 sy=.4 theta_deg=0 ag-dot-angle0.ctl | tee metal.out;
grep flux1: metal.out > metal.dat;
